function updateHobbiesList() {
  var hobbiesList = document.getElementById("hobbiesList");
  hobbiesList.innerHTML = "";

function deleteHobbyByValue(hobbyToDelete) {
  var index = profiles.hobbies.indexOf(hobbyToDelete);
  if (index !== -1) {
    profiles.hobbies.splice(index, 1);
    updateHobbiesList();
  }
}

function deleteHobby() {
  var hobbyToDelete = document.getElementById("deleteHobbyInput").value;
  deleteHobbyByValue(hobbyToDelete);
  document.getElementById("deleteHobbyInput").value = "";
}
